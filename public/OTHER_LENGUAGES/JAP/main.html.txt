<script>
    function changeSiteLanguage(lang) {
        const fileMap = {
            en: "../../main.html",
            jp: "../JAP/main.html",
            cn: "../CHI/main.html",
            es: "../ESP/main.html"
        };
        if (fileMap[lang]) {
            window.location.href = fileMap[lang];
        }
    }

    document.addEventListener("DOMContentLoaded", () => {
        const el = document.getElementById("hero-subtitle");
        const text = el.textContent;
        let i = 0;
        el.textContent = "";

        function type() {
            if (i < text.length) {
                el.innerHTML = text.substring(0, i + 1) +
                    '<span class="typewriter-cursor"></span>';
                i++;
                setTimeout(type, 50);
            } else {
                el.textContent = text;
            }
        }
        type();
    });
</script>
