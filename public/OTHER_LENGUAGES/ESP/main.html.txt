<script>
    function changeSiteLanguage(lang) {
        const pages = {
            en: "../../main.html",
            es: "main.html",
            jp: "../JAP/main.html",
            cn: "../CHI/main.html"
        };

        if (pages[lang]) {
            window.location.href = pages[lang];
        }
    }

    document.addEventListener("DOMContentLoaded", () => {
        const el = document.getElementById("hero-subtitle");
        const text = el.textContent;
        let i = 0;

        el.textContent = "";

        function type() {
            if (i < text.length) {
                el.innerHTML =
                    text.substring(0, i + 1) +
                    '<span class="typewriter-cursor"></span>';
                i++;
                setTimeout(type, 50);
            } else {
                el.textContent = text;
            }
        }

        type();
    });
</script>
